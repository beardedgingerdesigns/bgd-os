# AIOS UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local Next.js project tracker for Justin's AIOS that reads markdown across his existing repos, renders a triage-first dashboard per project, and provides a per-project chat panel that shells out to Claude Code with project context auto-loaded.

**Architecture:** Single-process Next.js 15 app (App Router). Server-side reads markdown from configured filesystem roots, parses YAML frontmatter, runs `claude` CLI as a subprocess for extraction / capture / chat. The filesystem is the source of truth — no database. A `chokidar` watcher invalidates an in-process extraction cache on file change and pushes Server-Sent Events to the client. Chat starts as one-shot `claude --print` per message; streaming is a follow-up enhancement.

**Tech Stack:** Next.js 15 (App Router), TypeScript, Tailwind CSS v4, shadcn/ui, Vitest, gray-matter, chokidar, js-yaml. Node ≥ 22. Subprocess via `child_process.spawn`. Source data: `~/repos/claude-os/{references,decisions,context,connections.md}` and `~/.claude/projects/-Users-justinlobaito-repos-claude-os/memory/`.

**Spec reference:** `~/repos/claude-os/docs/superpowers/specs/2026-05-07-aios-ui-design.md`. Re-read before starting Task 1.

```ts
  name: z.string().min(1),
  bucket: z.enum(['paying', 'prospects', 'internal']),
  projects: z.array(projectSchema).default([]),

  clients: z.array(clientSchema),
});

export function loadRegistryFromFile(filePath: string): Registry {
  const raw = fs.readFileSync(filePath, 'utf8');
  const parsed = yaml.load(raw);
  const validated = registrySchema.parse(parsed);
  return validated;
```

```ts
import fs from 'node:fs';
import path from 'node:path';
import matter from 'gray-matter';
import type { ProjectFile, FileFrontmatter } from './types';

export function readMarkdownFile(absPath: string): ProjectFile {
  const raw = fs.readFileSync(absPath, 'utf8');
  const stat = fs.statSync(absPath);
  const parsed = matter(raw);
  return {
    absPath,
    relPath: absPath,
    mtime: stat.mtimeMs,
    frontmatter: (parsed.data ?? {}) as FileFrontmatter,
    body: parsed.content,
  };
}

export function scanMarkdownFiles(rootDir: string): ProjectFile[] {
  const out: ProjectFile[] = [];
  function walk(dir: string) {
    let entries: fs.Dirent[];
    try {
      entries = fs.readdirSync(dir, { withFileTypes: true });
    } catch {
      return;
    }
    for (const e of entries) {
      const p = path.join(dir, e.name);
      if (e.isDirectory()) {
        if (e.name.startsWith('.') || e.name === 'node_modules') continue;
        walk(p);
      } else if (e.isFile() && e.name.endsWith('.md')) {
        try {
          out.push(readMarkdownFile(p));
        } catch {
          // skip unreadable
        }
      }
    }
  }
  walk(rootDir);
  return out.map(f => ({ ...f, relPath: path.relative(rootDir, f.absPath) }));
}

export function filterFilesForProject(files: ProjectFile[], projectSlug: string): ProjectFile[] {
  return files.filter(f => f.frontmatter.project === projectSlug);
}

export function unfiledFiles(files: ProjectFile[]): ProjectFile[] {
  return files.filter(f => !f.frontmatter.project);
}
```

```ts
import fs from 'node:fs';
import path from 'node:path';
import type { ExtractedProject } from './types';

export class ExtractionCache {
  constructor(private dir: string) {
    fs.mkdirSync(path.join(dir, 'extraction'), { recursive: true });
  }

  private fileFor(slug: string): string {
    return path.join(this.dir, 'extraction', `${slug}.json`);
  }

  get(slug: string): ExtractedProject | null {
    const f = this.fileFor(slug);
    if (!fs.existsSync(f)) return null;
    try {
      return JSON.parse(fs.readFileSync(f, 'utf8')) as ExtractedProject;
    } catch {
      return null;
    }
  }

  set(slug: string, data: ExtractedProject): void {
    fs.writeFileSync(this.fileFor(slug), JSON.stringify(data, null, 2), 'utf8');
  }

  invalidate(slug: string): void {
    const f = this.fileFor(slug);
    if (fs.existsSync(f)) fs.unlinkSync(f);
  }

  isStale(slug: string, currentSources: { path: string; mtime: number }[]): boolean {
    const cached = this.get(slug);
    if (!cached) return true;
    const cachedByPath = new Map(cached.source_files.map(s => [s.path, s.mtime]));
    if (cached.source_files.length !== currentSources.length) return true;
    for (const cur of currentSources) {
      const prev = cachedByPath.get(cur.path);
      if (prev === undefined || cur.mtime > prev) return true;
    }
    return false;
  }
}
```

```ts
Rules:
- action_items: pull from "What's blocking", "Open items", "Action items", or any prose-stated commitment ("I will...", "Justin to..."). Mark commitments as "open" by default.
- status_flags: pull from any risk language (overdue, past due, delinquent, incident, blocker, escalate). Pick severity: high for active incidents/AR/security, med for stale items, low for informational.
- recent_decisions: pull dated decisions, most recent 5.
- stakeholders: pull from any "Stakeholders" table or named contacts.
- next_dates: any future-dated commitment (calls, deadlines, launches).
- Return ONLY the JSON object. No prose. No markdown fences. No explanation.

Files for project follow.`;

export function parseExtractionOutput(raw: string): ExtractedProject | null {
  let trimmed = raw.trim();
  // Strip code fences if present
  const fenceMatch = trimmed.match(/^```(?:json)?\s*([\s\S]*?)\s*```$/);
  if (fenceMatch) trimmed = fenceMatch[1];
  // Find first { and last }
  const first = trimmed.indexOf('{');
  const last = trimmed.lastIndexOf('}');
  if (first === -1 || last === -1) return null;
  const candidate = trimmed.slice(first, last + 1);
  try {
    return JSON.parse(candidate) as ExtractedProject;
  } catch {
    return null;
  }
}

interface ExtractInput {
  projectSlug: string;
  files: ProjectFile[];
}

export async function extractProject(input: ExtractInput): Promise<ExtractedProject | null> {
  const filesPayload = input.files
    .map(f => `=== FILE: ${f.relPath} (mtime ${f.mtime}) ===\n${f.body}`)
    .join('\n\n');
  const prompt = `${EXTRACTION_PROMPT}\n\n${filesPayload}`;

  const out = await runClaudePrint(prompt, ['--model', 'claude-haiku-4-5-20251001']);
  let parsed = parseExtractionOutput(out);
  if (!parsed) {
    // retry once with a stricter directive
    const retryPrompt = `${prompt}\n\nIMPORTANT: Return ONLY a single JSON object. Do not wrap it in markdown fences. Do not add any prose before or after.`;
    const retry = await runClaudePrint(retryPrompt, ['--model', 'claude-haiku-4-5-20251001']);
    parsed = parseExtractionOutput(retry);
  }
  if (parsed) {
    parsed.source_files = input.files.map(f => ({ path: f.relPath, mtime: f.mtime }));
    parsed.generated_at = Date.now();
  }
  return parsed;
}

export function runClaudePrint(prompt: string, extraArgs: string[] = []): Promise<string> {
  return new Promise((resolve, reject) => {
    const proc = spawn(paths.claudeBin(), ['--print', ...extraArgs], {
      stdio: ['pipe', 'pipe', 'pipe'],
    });
    let stdout = '';
    let stderr = '';
    proc.stdout.on('data', d => (stdout += d.toString()));
    proc.stderr.on('data', d => (stderr += d.toString()));
    proc.on('error', reject);
    proc.on('close', code => {
      if (code !== 0) reject(new Error(`claude exited ${code}: ${stderr}`));
      else resolve(stdout);
    });
    proc.stdin.write(prompt);
    proc.stdin.end();
  });
}
```

```ts
import path from 'node:path';
import { loadRegistryFromFile } from '@/lib/registry';
import { scanMarkdownFiles, filterFilesForProject } from '@/lib/files';
import { ExtractionCache } from '@/lib/cache';
import { extractProject } from '@/lib/extractor';

export async function GET(
  _req: NextRequest,
  { params }: { params: Promise<{ slug: string }> }
) {
  const { slug } = await params;
  try {
    const registry = loadRegistryFromFile(paths.registryFile());
    const client = registry.clients.find(c => c.projects.some(p => p.slug === slug));
    const project = client?.projects.find(p => p.slug === slug);
    if (!client || !project) {
      return NextResponse.json({ error: 'project not found' }, { status: 404 });
    }

    // Scan all configured roots and filter to this project
    const roots = [
      paths.claudeOsDir(),
      paths.memoryDir(),
    ];
    const allFiles = roots.flatMap(r => scanMarkdownFiles(r));
    const projectFiles = filterFilesForProject(allFiles, slug);

    // Cache + extract
    const cache = new ExtractionCache(paths.cacheDir());
    const sources = projectFiles.map(f => ({ path: f.relPath, mtime: f.mtime }));
    let extraction = cache.get(slug);
    if (!extraction || cache.isStale(slug, sources)) {
      extraction = await extractProject({ projectSlug: slug, files: projectFiles }) ?? extraction;
      if (extraction) cache.set(slug, extraction);
    }

    const brief = projectFiles.find(f => f.frontmatter.type === 'brief');
    return NextResponse.json({
      client: { slug: client.slug, name: client.name, bucket: client.bucket },
      project: { slug: project.slug, name: project.name, status: project.status, contract: project.contract },
      brief: brief ? { path: brief.relPath, body: brief.body } : null,
      files: projectFiles.map(f => ({ path: f.relPath, frontmatter: f.frontmatter })),
      extraction,
    });
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}
```

```ts
      return NextResponse.json({ error: 'project not found' }, { status: 404 });
    }
    const all = [paths.claudeOsDir(), paths.memoryDir()].flatMap(scanMarkdownFiles);
    const files = filterFilesForProject(all, slug);
    const brief = files.find(f => f.frontmatter.type === 'brief');
    const cache = new ExtractionCache(paths.cacheDir());
    const reply = await runChat({
      projectSlug: slug,
      clientName: client.name,
      projectName: project.name,
      brief,
      extraction: cache.get(slug),
      userMessage,
    });
    return NextResponse.json({ reply });
  } catch (err: any) {
    return NextResponse.json({ error: err.message }, { status: 500 });
  }
}
```

```js
#!/usr/bin/env node
/**
 * One-time helper to add `client` and `project` frontmatter to existing markdown
 * files. Reads the same clients.yaml the app uses, then proposes a mapping based
 * on filename heuristics. Edits with --apply; previews otherwise.
 *
 * Usage:
 *   node scripts/add-frontmatter.mjs                   # dry-run (preview only)
 *   node scripts/add-frontmatter.mjs --apply           # write changes
 */
import fs from 'node:fs';
import path from 'node:path';
import yaml from 'js-yaml';
import matter from 'gray-matter';

const APPLY = process.argv.includes('--apply');
const CLAUDE_OS = process.env.CLAUDE_OS_DIR || '/Users/justinlobaito/repos/claude-os';
const MEMORY = process.env.CLAUDE_MEMORY_DIR || '/Users/justinlobaito/.claude/projects/-Users-justinlobaito-repos-claude-os/memory';

const registry = yaml.load(fs.readFileSync(path.join(CLAUDE_OS, 'clients.yaml'), 'utf8'));

// Build heuristic map: token -> {client, project}
const heuristics = [];
for (const c of registry.clients) {
  for (const p of c.projects) {
    heuristics.push({
      patterns: [c.slug, p.slug, c.name.toLowerCase().replace(/\s+/g, '-')],
      client: c.slug,
      project: p.slug,
    });
  }
}
// Aliases the slugs alone don't catch
const aliases = {
  insideout: 'inside-out-website',
  'inside-out-iowa': 'inside-out-website',
  'wild-rose': 'wild-rose-redesign',
  thermalkitchen: 'thermal-kitchen',
  'thermal-kitchen': 'thermal-kitchen',
  tonequest: 'tonequest-replatform',
  'tone-quest': 'tonequest-replatform',
  terraplex: 'terraplex-hub',
  brandos: 'brandos',
  russell: 'brandos-phase1',
  revolution: 'brandos-phase1',
  odebolt: 'terraplex-hub',
};

function classify(filename) {
  const lower = filename.toLowerCase();
  for (const [alias, projectSlug] of Object.entries(aliases)) {
    if (lower.includes(alias)) {
      const target = registry.clients.flatMap(c =>
        c.projects.map(p => ({ client: c.slug, project: p.slug }))
      ).find(x => x.project === projectSlug);
      if (target) return target;
    }
  }
  for (const h of heuristics) {
    if (h.patterns.some(p => lower.includes(p))) {
      return { client: h.client, project: h.project };
    }
  }
  return null;
}

function detectType(filename, body) {
  const lower = filename.toLowerCase();
  if (lower.includes('decision') || lower === 'log.md') return 'decision';
  if (lower.includes('memory_') || lower.startsWith('project_') || lower.startsWith('user_') || lower.startsWith('feedback_') || lower.startsWith('reference_')) return 'memory';
  if (body.length > 1500) return 'brief';
  return 'note';
}

const targets = [];
const SKIP_DIRS = new Set(['node_modules', 'archives', 'audits', 'docs']);
function walk(dir) {
  for (const e of fs.readdirSync(dir, { withFileTypes: true })) {
    const p = path.join(dir, e.name);
    if (e.isDirectory()) {
      if (e.name.startsWith('.') || SKIP_DIRS.has(e.name)) continue;
      walk(p);
    } else if (e.isFile() && e.name.endsWith('.md')) {
      targets.push(p);
    }
  }
}
walk(CLAUDE_OS);
walk(MEMORY);

let added = 0;
let skipped = 0;
const unmatched = [];

for (const file of targets) {
  if (file.includes('/node_modules/')) continue;
  const raw = fs.readFileSync(file, 'utf8');
  const parsed = matter(raw);
  if (parsed.data.client && parsed.data.project) {
    skipped++;
    continue;
  }
  const cls = classify(path.basename(file));
  if (!cls) {
    unmatched.push(file);
    continue;
  }
  const newData = {
    ...parsed.data,
    client: parsed.data.client ?? cls.client,
    project: parsed.data.project ?? cls.project,
    type: parsed.data.type ?? detectType(path.basename(file), parsed.content),
  };
  const newContent = matter.stringify(parsed.content, newData);
  if (APPLY) {
    fs.writeFileSync(file, newContent, 'utf8');
  } else {
    console.log(`[would write] ${file} -> client=${newData.client} project=${newData.project} type=${newData.type}`);
  }
  added++;
}

console.log(`\nSummary: ${added} updated, ${skipped} already had frontmatter, ${unmatched.length} unmatched.`);
if (unmatched.length) {
  console.log('\nUnmatched files (need manual frontmatter):');
  unmatched.forEach(f => console.log(`  ${f}`));
}
if (!APPLY) console.log('\nDry-run only. Re-run with --apply to write changes.');
```

```ts
import chokidar from 'chokidar';
import { EventEmitter } from 'node:events';
import { readMarkdownFile } from './files';
import { ExtractionCache } from './cache';
import { paths } from './paths';

class WatcherSingleton {
  private bus = new EventEmitter();
  private started = false;

  start() {
    if (this.started) return;
    this.started = true;
    const cache = new ExtractionCache(paths.cacheDir());
    const watcher = chokidar.watch(
      [
        `${paths.claudeOsDir()}/references/**/*.md`,
        `${paths.claudeOsDir()}/decisions/**/*.md`,
        `${paths.claudeOsDir()}/context/**/*.md`,
        `${paths.claudeOsDir()}/connections.md`,
        `${paths.claudeOsDir()}/clients.yaml`,
        `${paths.memoryDir()}/**/*.md`,
      ],
      { ignoreInitial: true, persistent: true }
    );
    watcher.on('all', (_event, filePath) => {
      let project: string | undefined;
      try {
        if (filePath.endsWith('.md')) {
          const f = readMarkdownFile(filePath);
          project = f.frontmatter.project;
        }
      } catch {
        // ignore
      }
      if (project) {
        cache.invalidate(project);
        this.bus.emit('invalidate', { project });
      } else {
        this.bus.emit('invalidate', { project: '*' });
      }
    });
  }

  on(event: 'invalidate', listener: (msg: { project: string }) => void) {
    this.bus.on(event, listener);
  }
  off(event: 'invalidate', listener: (msg: { project: string }) => void) {
    this.bus.off(event, listener);
  }
}

export const watcher = new WatcherSingleton();
```

```ts
export const runtime = 'nodejs';

export async function GET(_req: NextRequest) {
  watcher.start();
  const stream = new ReadableStream({
    start(controller) {
      const enc = new TextEncoder();
      const send = (data: { project: string }) => {
        controller.enqueue(enc.encode(`data: ${JSON.stringify(data)}\n\n`));
      };
      const listener = (msg: { project: string }) => send(msg);
      watcher.on('invalidate', listener);
      // periodic comment to keep connection alive
      const ka = setInterval(() => controller.enqueue(enc.encode(': ka\n\n')), 30_000);
      // cleanup on close
      const close = () => {
        clearInterval(ka);
        watcher.off('invalidate', listener);
      };
      // @ts-expect-error - using close hook
      controller.signal?.addEventListener?.('abort', close);
    },
  });
  return new Response(stream, {
    headers: {
      'content-type': 'text/event-stream',
      'cache-control': 'no-cache, no-transform',
      connection: 'keep-alive',
    },
  });
```
