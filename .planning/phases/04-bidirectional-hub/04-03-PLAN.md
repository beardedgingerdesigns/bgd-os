       ```tsx
       import { readWikiDecisions, readWikiLogEntries } from '@/lib/data/wiki'
       import { Badge } from '@/components/ui/badge'
       import { ExternalLink, BookOpen } from 'lucide-react'
       import path from 'path'

       interface WikiDisplayProps { wikiPath: string }

       export async function WikiDisplay({ wikiPath }: WikiDisplayProps) {
         const [decisions, logEntries] = await Promise.all([
           readWikiDecisions(wikiPath),
           readWikiLogEntries(wikiPath, { limit: 5 }),
         ])
         const wikiName = path.basename(wikiPath)
         return (
           <section className="mb-10">
             <div className="flex items-baseline justify-between mb-3">
               <h2 className="text-xs uppercase tracking-wider text-muted-foreground font-medium flex items-center gap-2">
                 <BookOpen className="h-3.5 w-3.5" /> Wiki — {wikiName}
               </h2>
               <a
                 href={`vscode://file/${wikiPath}`}
                 className="text-xs text-muted-foreground hover:text-foreground inline-flex items-center gap-1"
               >
                 Open wiki folder <ExternalLink className="h-3 w-3" />
               </a>
             </div>

             {/* Active decisions */}
             <details open className="mb-4 border-y border-border">
               <summary className="cursor-pointer py-3 flex items-center gap-2 font-medium">
                 Active decisions <Badge variant="secondary">{decisions.active.length}</Badge>
               </summary>
               <ul className="divide-y divide-border">
                 {decisions.active.length === 0 && (
                   <li className="py-3 text-sm text-muted-foreground">No active decisions in this wiki.</li>
                 )}
                 {decisions.active.map(d => (
                   <li key={d.filePath} className="py-3 flex items-start justify-between gap-4">
                     <div className="flex-1 min-w-0">
                       <div className="font-semibold text-sm">{d.title}</div>
                       {d.firstParagraph && (
                         <div className="text-sm text-muted-foreground line-clamp-2 mt-1">{d.firstParagraph}</div>
                       )}
                     </div>
                     <a
                       href={`vscode://file/${d.filePath}`}
                       className="shrink-0 text-xs text-muted-foreground hover:text-foreground inline-flex items-center gap-1"
                     >
                       Open in editor <ExternalLink className="h-3 w-3" />
                     </a>
                   </li>
                 ))}
               </ul>
             </details>

             {/* Recent log entries */}
             <details open className="mb-4 border-b border-border">
               <summary className="cursor-pointer py-3 flex items-center gap-2 font-medium">
                 Recent log entries <Badge variant="secondary">{logEntries.length}</Badge>
               </summary>
               <ul className="divide-y divide-border">
                 {logEntries.length === 0 && (
                   <li className="py-3 text-sm text-muted-foreground">No log entries.</li>
                 )}
                 {logEntries.map(e => (
                   <li key={`${e.path}-${e.date}-${e.title}`} className="py-3 flex items-start justify-between gap-4">
                     <div className="flex items-baseline gap-3 min-w-0">
                       <span className="font-mono text-xs text-muted-foreground shrink-0">{e.date}</span>
                       <span className="text-sm truncate">{e.title}</span>
                     </div>
                     <a
                       href={`vscode://file/${e.path}`}
                       className="shrink-0 text-xs text-muted-foreground hover:text-foreground inline-flex items-center gap-1"
                     >
                       Open <ExternalLink className="h-3 w-3" />
                     </a>
                   </li>
                 ))}
               </ul>
             </details>

             {/* Deferred decisions */}
             <details className="border-b border-border">
               <summary className="cursor-pointer py-3 flex items-center gap-2 font-medium">
                 Deferred decisions <Badge variant="secondary">{decisions.deferred.length}</Badge>
               </summary>
               <ul className="divide-y divide-border">
                 {decisions.deferred.length === 0 && (
                   <li className="py-3 text-sm text-muted-foreground">No deferred decisions.</li>
                 )}
                 {decisions.deferred.map(d => (
                   <li key={d.filePath} className="py-3 flex items-start justify-between gap-4">
                     <div className="flex-1 min-w-0">
                       <div className="font-semibold text-sm">{d.title}</div>
                       {d.firstParagraph && (
                         <div className="text-sm text-muted-foreground line-clamp-2 mt-1">{d.firstParagraph}</div>
                       )}
                     </div>
                     <a
                       href={`vscode://file/${d.filePath}`}
                       className="shrink-0 text-xs text-muted-foreground hover:text-foreground inline-flex items-center gap-1"
                     >
                       Open <ExternalLink className="h-3 w-3" />
                     </a>
                   </li>
                 ))}
               </ul>
             </details>
           </section>
         )
       }
       ```
