       ```typescript
       import { spawn } from 'child_process'

       // Reuse the extractTextDelta helper — copy from daily-ingest.ts inline (small).
       function extractTextDelta(line: string): string | null { /* identical to daily-ingest.ts */ }

       interface GmailContextOpts {
         projectName: string
         contacts: string[]
         daysBack?: number
         claudeBin?: string
         timeoutMs?: number
       }

       export async function fetchGmailContext(opts: GmailContextOpts): Promise<string> {
         const daysBack = opts.daysBack ?? 7
         const claudeBin = opts.claudeBin ?? 'claude'
         const timeoutMs = opts.timeoutMs ?? 30_000
         // Filter @-domain wildcards into 'or' for the prompt; pass plain emails as-is.
         const contactList = opts.contacts.length === 0
           ? '(no contacts on file)'
           : opts.contacts.join(', ')
         const prompt = `Search Gmail for threads from the last ${daysBack} days involving any of these participants: ${contactList}. For each thread, return one Markdown table row: | Thread ID | Subject | Last sender | Last message (ISO) | 1-line summary |. Header: | Thread ID | Subject | Last sender | Last message | Summary |. If no threads match, return exactly: "No recent Gmail threads for ${opts.projectName}."`

         const args = ['--print','--permission-mode','bypassPermissions','--output-format','stream-json','--include-partial-messages','--verbose', prompt]

         return new Promise<string>(resolve => {
           let lineBuffer = ''
           let aggregated = ''
           let settled = false
           const child = spawn(claudeBin, args, { shell: false, stdio: ['ignore','pipe','pipe'] })
           const timer = setTimeout(() => { if (!settled) { settled = true; child.kill('SIGKILL'); resolve('') } }, timeoutMs)
           child.stdout.on('data', d => {
             lineBuffer += d.toString()
             const lines = lineBuffer.split('\n')
             lineBuffer = lines.pop() ?? ''
             for (const line of lines) {
               const delta = extractTextDelta(line)
               if (delta !== null) aggregated += delta
             }
           })
           child.on('error', () => { if (!settled) { settled = true; clearTimeout(timer); resolve('') } })
           child.on('close', code => {
             if (settled) return
             settled = true
             clearTimeout(timer)
             if (lineBuffer.trim()) {
               const delta = extractTextDelta(lineBuffer)
               if (delta !== null) aggregated += delta
             }
             resolve(code === 0 ? aggregated : '')
           })
         })
       }
       ```

       ```typescript
       import fs from 'fs/promises'
       import { briefPathFor, buildBriefFor } from '@/lib/indexer/build-brief'
       import { fetchGmailContext } from '@/lib/skills/gmail-context'
       import { fetchCalendarContext } from '@/lib/skills/calendar-context'
       import { getProject, getClient } from '@/lib/data/clients'

       export interface BriefRead {
         brief: string
         builtAt: Date
         source: 'cache' | 'lazy-build' | 'fallback'
       }

       export async function readBriefOrBuild(clientSlug: string, projectSlug: string): Promise<BriefRead> {
         const filePath = briefPathFor(clientSlug, projectSlug)
         try {
           const [brief, stat] = await Promise.all([fs.readFile(filePath, 'utf-8'), fs.stat(filePath)])
           return { brief, builtAt: stat.mtime, source: 'cache' }
         } catch {
           // miss → lazy build
           const built = await buildBriefFor(clientSlug, projectSlug)
           const brief = await fs.readFile(built.filePath, 'utf-8').catch(() => '')
           return {
             brief,
             builtAt: new Date(built.builtAt),
             source: built.status === 'success' ? 'lazy-build' : 'fallback',
           }
         }
       }

       export interface LiveContext { gmail: string; calendar: string; fetchedAt: Date }

       export async function buildLiveContext(clientSlug: string, projectSlug: string): Promise<LiveContext> {
         const project = await getProject(clientSlug, projectSlug)
         const projectName = project?.name ?? projectSlug
         const contacts = (project?.contacts ?? []).filter(c => !c.startsWith('@'))   // plain emails only for the prompt
         const [gmail, calendar] = await Promise.all([
           fetchGmailContext({ projectName, contacts }).catch(() => ''),
           fetchCalendarContext({ projectName, contacts }).catch(() => ''),
         ])
         return { gmail, calendar, fetchedAt: new Date() }
       }

       export function assembleSeedPrompt(args: {
         brief: string
         gmail: string
         calendar: string
         projectLabel: string
       }): string {
         return [
           `# ${args.projectLabel} — context bundle`,
           '',
           '## Project brief',
           args.brief.trim() || '_No brief available._',
           '',
           '## Recent Gmail threads (last 7d)',
           args.gmail.trim() || 'No recent Gmail threads.',
           '',
           '## Upcoming calendar events (next 7d)',
           args.calendar.trim() || 'No upcoming events.',
           '',
           '---',
           '',
           `I'm working on ${args.projectLabel}. Briefly orient yourself using the above and offer to help. Keep your initial response short — I'll ask follow-ups.`,
         ].join('\n')
       }
       ```
